# Cheat sheet for Lilidog Openbox added keybinds.
# Note - "super" and "mod4" both represent the windows key.

Keybinds List =      super + F1
Refresh sxhkd keys = super + Escape
Wallpapers =         super + Shift + v
Random BG=           super + v
Dmenu =		         super + F3
Editor =             super + e
File Manager =       super + t
Gammy =              super + g    
Jgmenu =             super + j
Logout =             super + x
Logout-alt =         super + shift + x
hotcorners =         super + z
Screenshot =         PrtSctr
Scrot      =         alt + Print
Terminal =           super + return
Toggle tint2 =       super + w
Tint2 Options =      super + Shift + w
Thunar split view =  Alt + s

#Volume Controls
XF86AudioLowerVolume
	 pamixer --decrease 5 && notify-send "Volume down" -u low -t 500 
XF86AudioRaiseVolume
	 pamixer --increase 5 && notify-send "Volume up" -u low -t 500
XF86AudioMute
	 pamixer --toggle-mute && notify-send "Mute/Unmute" -u low -t 2000

#Brightness controls
XF86MonBrightnessDown
	brightnessctl s 10%-

XF86MonBrightnessUp
	brightnessctl s 10%+
